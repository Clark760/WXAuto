# M5 障碍物错位修正

> **涉及文件**：`hex_grid.gd`、`obstacle_manager.gd`

---

## 根因

障碍物可视化当前方案是：在 `spawn_obstacles` 时调 `hex_grid.get_hex_points_local(cell)` 获取六角形顶点 → 创建独立的 `Polygon2D` 挂到 `ObstacleLayer`。这些顶点是用**当时的 `hex_size` 和 `origin_offset`** 计算的硬编码值。

战斗开始时 `_apply_stage_grid_config()`（battlefield.gd 行 1146-1175）可能修改 `hex_grid.hex_size`（行 1157），之后 hex_grid 调用 `queue_redraw()` 用新 hex_size 重绘格子。但 `ObstacleLayer` 中已创建的 `Polygon2D` 的 `polygon` 顶点仍是旧值 → **格子变了，障碍物没跟上** → 错位。

---

## 推荐方案：直接在 hex_grid._draw() 中绘制障碍物格子

**不再创建独立 Polygon2D**，而是让 `hex_grid` 持有一个"被阻挡的格子集合 + 颜色"，在 `_draw()` 中直接对这些格子用不同颜色填充。这样无论 hex_size 或 origin_offset 如何变化，障碍物始终与格子同步。

### 修改 1：`hex_grid.gd` — 新增障碍物格子标记

```gdscript
# hex_grid.gd — 新增变量（在 @export 区域之后）：
var _obstacle_cells: Dictionary = {}  # cell_key(int) -> Color

func set_obstacle_cells(cells: Dictionary) -> void:
    # cells 格式: { cell_key_int: Color, ... }
    _obstacle_cells = cells
    queue_redraw()

func clear_obstacle_cells() -> void:
    _obstacle_cells.clear()
    queue_redraw()
```

### 修改 2：`hex_grid.gd` — `_draw()` 中绘制障碍物颜色（行 49-63）

```gdscript
# hex_grid.gd — _draw() 修改：
func _draw() -> void:
    var font: Font = ThemeDB.fallback_font
    var font_size: int = ThemeDB.fallback_font_size
    for r in range(grid_height):
        for q in range(grid_width):
            var cell: Vector2i = Vector2i(q, r)
            var center: Vector2 = axial_to_local(cell)
            var points: PackedVector2Array = _build_hex_points(center)

            # 障碍物格子用障碍物颜色填充，普通格子用默认颜色
            var cell_key: int = ((q & 0xFFFF) << 16) | (r & 0xFFFF)
            if _obstacle_cells.has(cell_key):
                draw_colored_polygon(points, _obstacle_cells[cell_key])
            else:
                draw_colored_polygon(points, fill_color)

            var outline: PackedVector2Array = points.duplicate()
            outline.append(points[0])
            draw_polyline(outline, line_color, 1.2, true)
            if draw_coordinates and font != null:
                var text: String = "%d,%d" % [q, r]
                draw_string(font, center + Vector2(-14, 4), text, HORIZONTAL_ALIGNMENT_LEFT, -1, font_size, coordinate_color)
```

### 修改 3：`obstacle_manager.gd` — 改用 hex_grid 标记而非独立 Polygon2D

```gdscript
# obstacle_manager.gd — 删除 _spawn_visual、_visual_layer、configure_visual_parent
# 新增 apply_visuals_to_grid：

func apply_visuals_to_grid(hex_grid: Node) -> void:
    # 将障碍物颜色信息注入 hex_grid，由 hex_grid._draw() 统一绘制
    if hex_grid == null or not is_instance_valid(hex_grid):
        return
    if not hex_grid.has_method("set_obstacle_cells"):
        return
    var cells_colors: Dictionary = {}
    for obstacle in _obstacles:
        var obstacle_type: String = str(obstacle.get("type", "rock"))
        var color: Color = OBSTACLE_COLORS.get(obstacle_type, Color(0.62, 0.42, 0.24, 0.72))
        var cells_value: Variant = obstacle.get("cells", [])
        if not (cells_value is Array):
            continue
        for cell_value in (cells_value as Array):
            if cell_value is Vector2i:
                var cell: Vector2i = cell_value as Vector2i
                cells_colors[_cell_key_int(cell)] = color
    hex_grid.call("set_obstacle_cells", cells_colors)

func clear_visuals_from_grid(hex_grid: Node) -> void:
    if hex_grid != null and is_instance_valid(hex_grid) and hex_grid.has_method("clear_obstacle_cells"):
        hex_grid.call("clear_obstacle_cells")
```

`spawn_obstacles` 末尾改为调用 `apply_visuals_to_grid(hex_grid)` 而非 `_spawn_visual`：

```gdscript
# obstacle_manager.gd — spawn_obstacles 修改：
func spawn_obstacles(obstacles: Array, hex_grid: Node) -> void:
    clear_all_obstacles()
    if hex_grid == null or not is_instance_valid(hex_grid):
        return
    for obstacle_value in obstacles:
        # ... 原有的 cells 解析和 _blocked_cells 注册逻辑不变 ...
        # 删除: _spawn_visual(hex_grid, obstacle_type, cell) 
    # 新增：统一注入到 hex_grid
    apply_visuals_to_grid(hex_grid)
```

`clear_all_obstacles` 同步清除：

```gdscript
func clear_all_obstacles() -> void:
    _obstacles.clear()
    _blocked_cells.clear()
    # 不再需要清除 _visual_layer 子节点
```

---

## 方案优势

| 对比项 | 旧方案（独立 Polygon2D） | 新方案（hex_grid._draw()） |
|--------|--------------------------|----------------------------|
| hex_size 变化时 | ❌ 需要重新计算所有顶点 | ✅ 自动用新 hex_size 绘制 |
| origin_offset 变化时 | ❌ 需要更新 position | ✅ 自动跟随 |
| Node 数量 | 障碍物数量 × Polygon2D | 0 额外节点 |
| 性能 | 每个障碍物一个独立节点 | 合并在 hex_grid._draw 中 |
| 代码量 | `configure_visual_parent` + `_spawn_visual` | `set_obstacle_cells` + `_draw` 中加 3 行 |

---

## 验证

1. 准备阶段：障碍物格子应与六角格完美对齐
2. 点击开始战斗（hex_size 可能变化）：障碍物仍然对齐
3. 多关卡切换（不同 hex_size 配置）：障碍物始终正确
